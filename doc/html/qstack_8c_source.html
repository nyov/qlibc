157 qstack_t *qstack(int options) {
158  qstack_t *stack = (qstack_t *) malloc(sizeof(qstack_t));
159  if (stack == NULL) {
160  errno = ENOMEM;
161  return NULL;
162  }
163 
164  memset((void *) stack, 0, sizeof(qstack_t));
165  stack->list = qlist(options);
166  if (stack->list == NULL) {
167  free(stack);
168  return NULL;
169  }
