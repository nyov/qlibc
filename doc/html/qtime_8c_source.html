202 time_t qtime_parse_gmtstr(const char *gmtstr) {
203  struct tm gmtm;
204  if (strptime(gmtstr, "%a, %d %b %Y %H:%M:%S", &gmtm) == NULL)
205  return 0;
206  time_t utc = timegm(&gmtm);
207  if (utc < 0)
208  return -1;
209 
210  // parse timezone
211  char *p;
212  if ((p = strstr(gmtstr, "+")) != NULL) {
213  utc -= ((atoi(p + 1) / 100) * 60 * 60);
214  if (utc < 0)
215  return -1;
216  } else if ((p = strstr(gmtstr, "-")) != NULL) {
217  utc += ((atoi(p + 1) / 100) * 60 * 60);
218  }
219 
220  return utc;
