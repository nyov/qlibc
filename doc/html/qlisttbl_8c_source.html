281 static bool put(qlisttbl_t *tbl, const char *name, const void *data, size_t size)
282 {
283  // make new object table
284  qdlnobj_t *obj = _createobj(name, data, size);
285  if (obj == NULL) {
286  return false;
287  }
288 
289  // lock table
290  lock(tbl);
291 
292  // if unique flag is set, remove same key
293  if (tbl->unique == true) remove_(tbl, name);
294 
295  // insert into table
296  if (tbl->num == 0) {
297  obj->prev = NULL;
298  obj->next = NULL;
299  } else {
300  if (tbl->inserttop == false) {
301  obj->prev = tbl->last;
302  obj->next = NULL;
303  } else {
304  obj->prev = NULL;
305  obj->next = tbl->first;
306  }
307  }
308  _insertobj(tbl, obj);
309 
310  // unlock table
311  unlock(tbl);
312 
313  return true;

533 static qobj_t *getmulti(qlisttbl_t *tbl, const char *name, bool newmem,
534  size_t *numobjs)
535 {
536  qobj_t *objs = NULL; // objects container
537  size_t allocobjs = 0; // allocated number of objs
538  size_t numfound = 0; // number of keys found
539 
540  qdlnobj_t obj;
541  memset((void *)&obj, 0, sizeof(obj)); // must be cleared before call
542  lock(tbl);
543  while (tbl->getnext(tbl, &obj, name, newmem) == true) {
544  numfound++;
545 
546  // allocate object array.
547  if (numfound >= allocobjs) {
548  if (allocobjs == 0) allocobjs = 10; // start from 10
549  else allocobjs *= 2; // double size
550  objs = (qobj_t *)realloc(objs, sizeof(qobj_t) * allocobjs);
551  if (objs == NULL) {
552  DEBUG("qlisttbl->getmulti(): Memory reallocation failure.");
553  errno = ENOMEM;
554  break;
555  }
556  }
557 
558  // copy reference
559  qobj_t *newobj = &objs[numfound - 1];
560  newobj->data = obj.data;
561  newobj->size = obj.size;
562  newobj->type = (newmem == false) ? 1 : 2;
563 
564  // release resource
565  if (newmem == true) {
566  if (obj.name != NULL) free(obj.name);
567  }
568 
569  // clear next block
570  newobj = &objs[numfound];
571  memset((void *)newobj, '\0', sizeof(qobj_t));
572  newobj->type = 0; // mark, end of objects
573  }
574  unlock(tbl);
575 
576  // return found counter
577  if (numobjs != NULL) {
578  *numobjs = numfound;
579  }
580 
581  if (numfound == 0) {
582  errno = ENOENT;
583  }
584 
585  return objs;
586 }

660 static bool getnext(qlisttbl_t *tbl, qdlnobj_t *obj, const char *name,
661  bool newmem)
662 {
663  if (obj == NULL) return NULL;
664 
665  lock(tbl);
666 
667  qdlnobj_t *cont = NULL;
668  if (obj->size == 0) { // first time call
669  if (name == NULL) { // full scan
670  cont = (tbl->lookupforward) ? tbl->first : tbl->last;
671  } else { // name search
672  cont = _findobj(tbl, name, NULL);
673  }
674  } else { // next call
675  cont = (tbl->lookupforward) ? obj->next : obj->prev;
676  }
677 
678  if (cont == NULL) {
679  errno = ENOENT;
680  unlock(tbl);
681  return false;
682  }
683 
684  uint32_t hash = (name != NULL) ? qhashmurmur3_32(name, strlen(name)) : 0;
685 
686  bool ret = false;
687  while (cont != NULL) {
688  if (name == NULL || tbl->namematch(cont, name, hash) == true) {
689  if (newmem == true) {
690  obj->name = strdup(cont->name);
691  obj->data = malloc(cont->size);
692  if (obj->name == NULL || obj->data == NULL) {
693  if (obj->name != NULL) free(obj->name);
694  if (obj->data != NULL) free(obj->data);
695  obj->name = NULL;
696  obj->data = NULL;
697  errno = ENOMEM;
698  break;
699  }
700  memcpy(obj->data, cont->data, cont->size);
701  } else {
702  obj->name = cont->name;
703  obj->data = cont->data;
704  }
705  obj->hash = cont->hash;
706  obj->size = cont->size;
707  obj->prev = cont->prev;
708  obj->next = cont->next;
709 
710  ret = true;
711  break;
712  }
713 
714  cont = (tbl->lookupforward) ? cont->next : cont->prev;
715  }
716  unlock(tbl);
717 
718  if (ret == false) {
719  errno = ENOENT;
720  }
721 
722  return ret;
723 }

776 static bool removeobj(qlisttbl_t *tbl, const qdlnobj_t *obj)
777 {
778  if (obj == NULL) return false;
779 
780  lock(tbl);
781 
782  // copy chains
783  qdlnobj_t *prev = obj->prev;
784  qdlnobj_t *next = obj->next;
785 
786  // find this object
787  qdlnobj_t *this = NULL;
788  if (prev != NULL) this = prev->next;
789  else if (next != NULL) this = next->prev;
790  else this = tbl->first; // table has only one object.
791 
792  // double check
793  if (this == NULL) {
794  unlock(tbl);
795  DEBUG("qlisttbl->removeobj(): Can't veryfy object.");
796  errno = ENOENT;
797  return false;
798  }
799 
800  // adjust chain links
801  if (prev == NULL) tbl->first = next; // if the object is first one
802  else prev->next = next; // not the first one
803 
804  if (next == NULL) tbl->last = prev; // if the object is last one
805  else next->prev = prev; // not the first one
806 
807  // adjust counter
808  tbl->num--;
809 
810  unlock(tbl);
811 
812  // free object
813  free(this->name);
814  free(this->data);
815  free(this);
816 
817  return true;
818 }

854 static void sort_(qlisttbl_t *tbl)
855 {
856  // run bubble sort
857  lock(tbl);
858  qdlnobj_t *obj1, *obj2;
859  qdlnobj_t tmpobj;
860  int n, n2, i;
861  for (n = tbl->num; n > 0;) {
862  n2 = 0;
863  for (i = 0, obj1 = tbl->first; i < (n - 1); i++, obj1 = obj1->next) {
864  obj2 = obj1->next; // this can't be null.
865  if (tbl->namecmp(obj1->name, obj2->name) > 0) {
866  // swapping contents is faster than adjusting links.
867  tmpobj = *obj1;
868  obj1->hash = obj2->hash;
869  obj1->name = obj2->name;
870  obj1->data = obj2->data;
871  obj1->size = obj2->size;
872  obj2->hash = tmpobj.hash;
873  obj2->name = tmpobj.name;
874  obj2->data = tmpobj.data;
875  obj2->size = tmpobj.size;
876 
877  n2 = i + 1;
878  }
879  }
880  n = n2; // skip sorted tailing elements
881  }
882  unlock(tbl);
883 }

970  bool decode)

974  if (str == NULL) return -1;
975 
976  // parse
977  lock(tbl);
978  char *offset, *buf;
979  int cnt = 0;
980  for (offset = str; *offset != '\0'; ) {
981  // get one line into buf
982  for (buf = offset; *offset != '\n' && *offset != '\0'; offset++);
983  if (*offset != '\0') {
984  *offset = '\0';
985  offset++;
986  }
987  qstrtrim(buf);
988

990  if ((buf[0] == '#') || (buf[0] == '\0')) continue;
991 
992  // parse
993  char *data = strdup(buf);
994  char *name = _q_makeword(data, sepchar);
995  qstrtrim(data);
996  qstrtrim(name);
997  if (decode == true) qurl_decode(data);
998

1000  put(tbl, name, data, strlen(data) + 1);
1001 
1002  free(name);
1003  free(data);
1004  }
1005  unlock(tbl);

1130 static qdlnobj_t *_findobj(qlisttbl_t *tbl, const char *name, qdlnobj_t *retobj)
1131 {
1132  if (retobj != NULL) {
1133  memset((void *)retobj, '\0', sizeof(qdlnobj_t));
1134  }
1135 
1136  if (name == NULL || tbl->num == 0) {
1137  errno = ENOENT;
1138  return NULL;
1139  }
1140 
1141  uint32_t hash = qhashmurmur3_32(name, strlen(name));
1142  qdlnobj_t *obj = (tbl->lookupforward) ? tbl->first : tbl->last;
1143  while (obj != NULL) {
1144  // name string will be compared only if the hash matches.
1145  if (tbl->namematch(obj, name, hash) == true) {
1146  if (retobj != NULL) {
1147  *retobj = *obj;
1148  }
1149  return obj;
1150  }
1151  obj = (tbl->lookupforward)? obj->next : obj->prev;
1152  }
1153 
1154  // not found, set prev and next chain.
