29 /**
30  * @file qvector.c Vector implementation.
31  *
32  * qvector container is a vector implementation. It implements a growable array
33  * of objects and it extends qlist container that allow a linked-list to be
34  * treated as a vector.
35  *
36  * @code
37  * [Code sample - Object]
38  * qvector_t *vector = qvector(QVECTOR_THREADSAFE);
39  *
40  * // add elements
41  * vector->addstr(vector, "AB"); // no need to supply size
42  * vector->addstrf(vector, "%d", 12); // for formatted string
43  * vector->addstr(vector, "CD");
44  *
45  * // get the chunk as a string
46  * char *final = vector->tostring(vector);
47  *
48  * // print out
49  * printf("Number of elements = %zu\n", vector->size(vector));
50  * printf("Final string = %s\n", final);
51  *
52  * // release
53  * free(final);
54  * vector->free(vector);
55  *
56  * [Result]
57  * Number of elements = 3
58  * Final string = AB12CD
59  * @endcode
60  *
61  * @code
62  * [Code sample - Object]
63  * // sample object
64  * struct sampleobj {
65  * int num;
66  * char str[10];
67  * };
68  *
69  * // get new vector
70  * qvector_t *vector = qvector();
71  *
72  * // add objects
73  * int i;
74  * struct sampleobj obj;
75  * for(i = 0; i < 3; i++) {
76  * // filling object with sample data
77  * obj.num = i;
78  * sprintf(obj.str, "hello%d", i);
79  *
80  * // stack
81  * vector->add(vector, (void *)&obj, sizeof(struct sampleobj));
82  * }
83  *
84  * // final
85  * struct sampleobj *final;
86  * final = (struct sampleobj *)vector->toarray(vector, NULL);
87  *
88  * // print out
89  * printf("Number of Objects = %zu\n", vector->size(vector));
90  * for(i = 0; i < vector->size(vector); i++) {
91  * printf("Object%d %d, %s\n", i+1, final[i].num, final[i].str);
92  * }
93  *
94  * // release
95  * free(final);
96  * vector->free(vector);
97  *
98  * [Result]
99  * Number of Objects = 3
100  * Object1 0, hello0
101  * Object2 1, hello1
102  * Object3 2, hello2
103  * @endcode
104  */

131 /**
132  * Initialize vector.
133  *
134  * @param options combination of initialization options.
135  *
136  * @return qvector_t container pointer.
137  * @retval errno will be set in error condition.
138  * - ENOMEM : Memory allocation failure.
139  *
140  * @code
141  * // allocate memory
142  * qvector_t *vector = qvector(QVECTOR_THREADSAFE);
143  * vector->free(vector);
144  * @endcode
145  *
146  * @note
147  * Available options:
148  * - QVECTOR_THREADSAFE - make it thread-safe.
149  */
150 qvector_t *qvector(int options) {
151  qvector_t *vector = (qvector_t *) calloc(1, sizeof(qvector_t));
152  if (vector == NULL) {
153  errno = ENOMEM;
154  return NULL;
155  }
156 
157  vector->list = qlist(options);
158  if (vector->list == NULL) {
159  free(vector);
160  errno = ENOMEM;
161  return NULL;
162  }
163 
164  // methods
165  vector->add = add;
166  vector->addstr = addstr;
167  vector->addstrf = addstrf;
168 
169  vector->toarray = toarray;
170  vector->tostring = tostring;
171 
172  vector->size = size;
173  vector->datasize = datasize;
174  vector->clear = clear;
175  vector->debug = debug;
176  vector->free = free_;
177 
178  return vector;
179 }

223 static bool addstrf(qvector_t *vector, const char *format, ...) {
224  char *str;
225  DYNAMIC_VSPRINTF(str, format);
226  if (str == NULL) {
227  errno = ENOMEM;
228  return false;
229  }
230 
231  bool ret = addstr(vector, str);
232  free(str);
233 
234  return ret;
235 }
