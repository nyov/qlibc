223 qhasharr_t *qhasharr(void *memory, size_t memsize) {
224  // Structure memory.
225  qhasharr_data_t *data = (qhasharr_data_t *) memory;
226 
227  // Initialize data if memsize is set or use existing data.
228  if (memsize > 0) {
229  // calculate max
230  int maxslots = (memsize - sizeof(qhasharr_data_t))
231  / sizeof(qhasharr_slot_t);
232  if (maxslots < 1 || memsize <= sizeof(qhasharr_t)) {
233  errno = EINVAL;
234  return NULL;
235  }
236 
237  // Set memory.
238  memset((void *) data, 0, memsize);
239  data->maxslots = maxslots;
240  data->usedslots = 0;
241  data->num = 0;
242  }
243 
244  // Set data address. Shared memory returns virtul address.
245  data->slots = (qhasharr_slot_t *) (memory + sizeof(qhasharr_data_t));
246 
247  // Create the table object.
248  qhasharr_t *tbl = (qhasharr_t *) malloc(sizeof(qhasharr_t));
249  if (tbl == NULL) {
250  errno = ENOMEM;
251  return NULL;
252  }
253  memset((void *) tbl, 0, sizeof(qhasharr_t));
254 
255  // assign methods

293 static bool put(qhasharr_t *tbl, const char *key, const void *value,
294  size_t size) {
295  if (tbl == NULL || key == NULL || value == NULL) {
296  errno = EINVAL;
297  return false;
298  }
299 
300  qhasharr_data_t *data = tbl->data;
301 
302  // check full
303  if (data->usedslots >= data->maxslots) {
304  DEBUG("hasharr: put %s - FULL", key);
305  errno = ENOBUFS;
306  return false;
307  }
308 
309  // get hash integer
310  unsigned int hash = qhashmurmur3_32(key, strlen(key)) % data->maxslots;
311 
312  // check, is slot empty
313  if (data->slots[hash].count == 0) { // empty slot
314  // put data
315  if (_put_data(tbl, hash, hash, key, value, size, 1) == false) {
316  DEBUG("hasharr: FAILED put(new) %s", key);
317  return false;
318  } DEBUG("hasharr: put(new) %s (idx=%d,hash=%u,tot=%d)",
319  key, hash, hash, data->usedslots);
320  } else if (data->slots[hash].count > 0) { // same key or hash collision
321  // check same key;
322  int idx = _get_idx(tbl, key, hash);
323  if (idx >= 0) { // same key
324  // remove and recall
325  remove_(tbl, key);
326  return put(tbl, key, value, size);
327  } else { // no same key, just hash collision
328  // find empty slot
329  int idx = _find_empty(tbl, hash);
330  if (idx < 0) {
331  errno = ENOBUFS;
332  return false;
333  }
334 
335  // put data. -1 is used for collision resolution (idx != hash);
336  if (_put_data(tbl, idx, hash, key, value, size, -1) == false) {
337  DEBUG("hasharr: FAILED put(col) %s", key);
338  return false;
339  }
340 
341  // increase counter from leading slot
342  data->slots[hash].count++;
343 
344  DEBUG("hasharr: put(col) %s (idx=%d,hash=%u,tot=%d)",
345  key, idx, hash, data->usedslots);
346  }
347  } else {
348  // in case of -1 or -2, move it. -1 used for collision resolution,
349  // -2 used for oversized value data.
350 
351  // find empty slot
352  int idx = _find_empty(tbl, hash + 1);
353  if (idx < 0) {
354  errno = ENOBUFS;
355  return false;
356  }
357 
358  // move dup slot to empty
359  _copy_slot(tbl, idx, hash);
360  _remove_slot(tbl, hash);
361 
362  // in case of -2, adjust link of mother
363  if (data->slots[idx].count == -2) {
364  data->slots[data->slots[idx].hash].link = idx;
365  if (data->slots[idx].link != -1) {
366  data->slots[data->slots[idx].link].hash = idx;
367  }
368  }
369 
370  // store data
371  if (_put_data(tbl, hash, hash, key, value, size, 1) == false) {
372  DEBUG("hasharr: FAILED put(swp) %s", key);
373  return false;
374  }
375 
376  DEBUG("hasharr: put(swp) %s (idx=%u,hash=%u,tot=%d)",
377  key, hash, hash, data->usedslots);
378  }
379 
380  return true;
381 }

611 static bool remove_(qhasharr_t *tbl, const char *key) {
612  if (tbl == NULL || key == NULL) {
613  errno = EINVAL;
614  return false;
615  }
616 
617  qhasharr_data_t *data = tbl->data;
618 
619  // get hash integer
620  unsigned int hash = qhashmurmur3_32(key, strlen(key)) % data->maxslots;
621 
622  int idx = _get_idx(tbl, key, hash);
623  if (idx < 0) {
624  DEBUG("not found %s", key);
625  errno = ENOENT;
626  return false;
627  }
628 
629  if (data->slots[idx].count == 1) {
630  // just remove
631  _remove_data(tbl, idx);
632  DEBUG("hasharr: rem %s (idx=%d,tot=%d)", key, idx, data->usedslots);
633  } else if (data->slots[idx].count > 1) { // leading slot and has dup
634  // find dup
635  int idx2;
636  for (idx2 = idx + 1;; idx2++) {
637  if (idx2 >= data->maxslots)
638  idx2 = 0;
639  if (idx2 == idx) {
640  DEBUG("hasharr: [BUG] failed to remove dup key %s.", key);
641  errno = EFAULT;
642  return false;
643  }
644  if (data->slots[idx2].count == -1
645  && data->slots[idx2].hash == hash) {
646  break;
647  }
648  }
649 
650  // move to leading slot
651  int backupcount = data->slots[idx].count;
652  _remove_data(tbl, idx); // remove leading data
653  _copy_slot(tbl, idx, idx2); // copy slot
654  _remove_slot(tbl, idx2); // remove moved slot
655 
656  data->slots[idx].count = backupcount - 1; // adjust collision counter
657  if (data->slots[idx].link != -1) {
658  data->slots[data->slots[idx].link].hash = idx;
659  }
660 
661  DEBUG("hasharr: rem(lead) %s (idx=%d,tot=%d)",
662  key, idx, data->usedslots);
663  } else { // in case of -1. used for collision resolution
664  // decrease counter from leading slot
665  if (data->slots[data->slots[idx].hash].count <= 1) {
666  DEBUG("hasharr: [BUG] failed to remove %s. "
667  "counter of leading slot mismatch.", key);
668  errno = EFAULT;
669  return false;
670  }
671  data->slots[data->slots[idx].hash].count--;
672 
673  // remove data
674  _remove_data(tbl, idx);
675  DEBUG("hasharr: rem(dup) %s (idx=%d,tot=%d)", key, idx, data->usedslots);
676  }
677 
678  return true;
679 }

842 static int _get_idx(qhasharr_t *tbl, const char *key, unsigned int hash) {
843  qhasharr_data_t *data = tbl->data;
844 
845  if (data->slots[hash].count > 0) {
846  int count, idx;
847  for (count = 0, idx = hash; count < data->slots[hash].count;) {
848  if (data->slots[idx].hash == hash
849  && (data->slots[idx].count > 0
850  || data->slots[idx].count == -1)) {
851  // same hash
852  count++;
853 
854  // is same key?
855  size_t keylen = strlen(key);
856  // first check key length
857  if (keylen == data->slots[idx].data.pair.keylen) {
858  if (keylen <= _Q_HASHARR_KEYSIZE) {
859  // original key is stored
860  if (!memcmp(key, data->slots[idx].data.pair.key,
861  keylen)) {
862  return idx;
863  }
864  } else {
865  // key is truncated, compare MD5 also.
866  unsigned char keymd5[16];
867  qhashmd5(key, keylen, keymd5);
868  if (!memcmp(key, data->slots[idx].data.pair.key,
869  _Q_HASHARR_KEYSIZE)
870  && !memcmp(keymd5,
871  data->slots[idx].data.pair.keymd5,
872  16)) {
873  return idx;
874  }
875  }
876  }
877  }
878 
879  // increase idx
880  idx++;
881  if (idx >= data->maxslots)
882  idx = 0;
883 
884  // check loop
885  if (idx == hash)
886  break;
887 
888  continue;
889  }
890  }
891 
892  return -1;
893 }

895 static void *_get_data(qhasharr_t *tbl, int idx, size_t *size) {
896  if (idx < 0) {
897  errno = ENOENT;
898  return NULL;
899  }
900 
901  qhasharr_data_t *data = tbl->data;
902 
903  int newidx;
904  size_t valsize;
905  for (newidx = idx, valsize = 0;; newidx = data->slots[newidx].link) {
906  valsize += data->slots[newidx].size;
907  if (data->slots[newidx].link == -1)
908  break;
909  }
910 
911  void *value, *vp;
912  value = malloc(valsize);
913  if (value == NULL) {
914  errno = ENOMEM;
915  return NULL;
916  }
917 
918  for (newidx = idx, vp = value;; newidx = data->slots[newidx].link) {
919  if (data->slots[newidx].count == -2) {
920  // extended data block
921  memcpy(vp, (void *) data->slots[newidx].data.ext.value,
922  data->slots[newidx].size);
923  } else {
924  // key/value pair data block
925  memcpy(vp, (void *) data->slots[newidx].data.pair.value,
926  data->slots[newidx].size);
927  }
928 
929  vp += data->slots[newidx].size;
930  if (data->slots[newidx].link == -1)
931  break;
932  }
933 
934  if (size != NULL)
935  *size = valsize;
936  return value;
937 }

939 static bool _put_data(qhasharr_t *tbl, int idx, unsigned int hash,
940  const char *key, const void *value, size_t size,
941  int count) {
942  qhasharr_data_t *data = tbl->data;
943 
944  // check if used
945  if (data->slots[idx].count != 0) {
946  DEBUG("hasharr: BUG found.");
947  errno = EFAULT;
948  return false;
949  }
950 
951  size_t keylen = strlen(key);
952  unsigned char keymd5[16];
953  qhashmd5(key, keylen, keymd5);
954 
955  // store key
956  data->slots[idx].count = count;
957  data->slots[idx].hash = hash;
958  strncpy(data->slots[idx].data.pair.key, key, _Q_HASHARR_KEYSIZE);
959  memcpy((char *) data->slots[idx].data.pair.keymd5, (char *) keymd5, 16);
960  data->slots[idx].data.pair.keylen = keylen;
961  data->slots[idx].link = -1;
962 
963  // store value
964  int newidx;
965  size_t savesize;
966  for (newidx = idx, savesize = 0; savesize < size;) {
967  if (savesize > 0) { // find next empty slot
968  int tmpidx = _find_empty(tbl, newidx + 1);
969  if (tmpidx < 0) {
970  DEBUG("hasharr: Can't expand slot for key %s.", key);
971  _remove_data(tbl, idx);
972  errno = ENOBUFS;
973  return false;
974  }
975 
976  // clear & set
977  memset((void *) (&data->slots[tmpidx]), '\0',
978  sizeof(qhasharr_slot_t));
979 
980  data->slots[tmpidx].count = -2; // extended data block
981  data->slots[tmpidx].hash = newidx; // prev link
982  data->slots[tmpidx].link = -1; // end block mark
983  data->slots[tmpidx].size = 0;
984 
985  data->slots[newidx].link = tmpidx; // link chain
