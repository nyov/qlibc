157 qqueue_t *qqueue(int options) {
158  qqueue_t *queue = (qqueue_t *) malloc(sizeof(qqueue_t));
159  if (queue == NULL) {
160  errno = ENOMEM;
161  return NULL;
162  }
163 
164  memset((void *) queue, 0, sizeof(qqueue_t));
165  queue->list = qlist(options);
166  if (queue->list == NULL) {
