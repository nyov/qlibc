312 static bool addat(qlist_t *list, int index, const void *data, size_t size) {
313  // check arguments
314  if (data == NULL || size <= 0) {
315  errno = EINVAL;
316  return false;
317  }
318 
319  lock(list);
320 
321  // check maximum number of allowed elements if set
322  if (list->max > 0 && list->num >= list->max) {
323  errno = ENOBUFS;
324  unlock(list);
325  return false;
326  }
327 
328  // adjust index
329  if (index < 0)
330  index = (list->num + index) + 1; // -1 is same as addlast()
331  if (index < 0 || index > list->num) {
332  // out of bound
333  unlock(list);
334  errno = ERANGE;
335  return false;
336  }
337 
338  // duplicate object
339  void *dup_data = malloc(size);
340  if (dup_data == NULL) {
341  unlock(list);
342  errno = ENOMEM;
343  return false;
344  }
345  memcpy(dup_data, data, size);
346 
347  // make new object list
348  qdlobj_t *obj = (qdlobj_t *) malloc(sizeof(qdlobj_t));
349  if (obj == NULL) {
350  free(dup_data);
351  unlock(list);
352  errno = ENOMEM;
353  return false;
354  }
355  obj->data = dup_data;
356  obj->size = size;
357  obj->prev = NULL;
358  obj->next = NULL;
359 
360  // make link
361  if (index == 0) {
362  // add at first
363  obj->next = list->first;
364  if (obj->next != NULL)
365  obj->next->prev = obj;
366  list->first = obj;
367  if (list->last == NULL)
368  list->last = obj;
369  } else if (index == list->num) {
370  // add after last
371  obj->prev = list->last;
372  if (obj->prev != NULL)
373  obj->prev->next = obj;
374  list->last = obj;
375  if (list->first == NULL)
376  list->first = obj;
377  } else {
378  // add at the middle of list
379  qdlobj_t *tgt = _get_obj(list, index);
380  if (tgt == NULL) {
381  // should not be happened.
382  free(dup_data);
383  free(obj);
384  unlock(list);
385  errno = EAGAIN;
386  return false;
387  }
388 
389  // insert obj
390  tgt->prev->next = obj;
391  obj->prev = tgt->prev;
392  obj->next = tgt;
393  tgt->prev = obj;
394  }
395 
396  list->datasum += size;
397  list->num++;
398 
399  unlock(list);
400 
401  return true;
402 }

784 static char *tostring(qlist_t *list) {
785  if (list->num <= 0) {
786  errno = ENOENT;
787  return NULL;
788  }
789 
790  lock(list);
791 
792  void *chunk = malloc(list->datasum + 1);
793  if (chunk == NULL) {
794  unlock(list);
795  errno = ENOMEM;
796  return NULL;
797  }
798  void *dp = chunk;
799 
800  qdlobj_t *obj;
801  for (obj = list->first; obj; obj = obj->next) {
802  size_t size = obj->size;
803  // do not copy tailing '\0'
804  if (*(char *) (obj->data + (size - 1)) == '\0')
805  size -= 1;
806  memcpy(dp, obj->data, size);
807  dp += size;
808  }
809  *((char *) dp) = '\0';
810  unlock(list);
811 
812  return (char *) chunk;
813 }

882  lock(list);
883 
884  // get object pointer
885  qdlobj_t *obj = _get_obj(list, index);
886  if (obj == NULL) {
887  unlock(list);
888  return false;
889  }
890 
891  // copy data
892  void *data;
893  if (newmem == true) {
894  data = malloc(obj->size);
895  if (data == NULL) {
896  unlock(list);
897  errno = ENOMEM;
898  return false;
899  }
900  memcpy(data, obj->data, obj->size);
901  } else {
902  data = obj->data;
903  }
904  if (size != NULL)
905  *size = obj->size;
906 
907  // remove if necessary
908  if (remove == true) {
909  if (_remove_obj(list, obj) == false) {
910  if (newmem == true)
911  free(data);
912  data = NULL;
913  }
914  }

921 static qdlobj_t *_get_obj(qlist_t *list, int index) {
922  // index adjustment
923  if (index < 0)
924  index = list->num + index;
925  if (index >= list->num) {
926  errno = ERANGE;
927  return NULL;
928  }
929 
930  // detect faster scan direction
931  bool backward;
932  qdlobj_t *obj;
933  int listidx;
934  if (index < list->num / 2) {
935  backward = false;
936  obj = list->first;
937  listidx = 0;
938  } else {
939  backward = true;
940  obj = list->last;
941  listidx = list->num - 1;
942  }
943 
944  // find object
945  while (obj != NULL) {
946  if (listidx == index)
947  return obj;
948 
949  if (backward == false) {
950  obj = obj->next;
951  listidx++;
952  } else {
953  obj = obj->prev;
954  listidx--;
955  }
956  }
957 
958  // never reach here
959  errno = ENOENT;
960  return NULL;
961 }

963 static bool _remove_obj(qlist_t *list, qdlobj_t *obj) {
964  if (obj == NULL)
965  return false;
966 
967  // chain prev and next elements
968  if (obj->prev == NULL)
969  list->first = obj->next;
970  else
971  obj->prev->next = obj->next;
972  if (obj->next == NULL)
973  list->last = obj->prev;
974  else
975  obj->next->prev = obj->prev;
976 
977  // adjust counter
978  list->datasum -= obj->size;
979  list->num--;
980 
981  // release obj
982  free(obj->data);
983  free(obj);
984 
985  return true;
986 }
