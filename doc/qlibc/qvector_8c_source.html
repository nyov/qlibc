150 qvector_t *qvector(int options) {
151  qvector_t *vector = (qvector_t *) calloc(1, sizeof(qvector_t));
152  if (vector == NULL) {
153  errno = ENOMEM;
154  return NULL;
155  }
156 
157  vector->list = qlist(options);
158  if (vector->list == NULL) {
159  free(vector);

161  return NULL;
